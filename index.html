Convert number to reversed array of digits

                        
    function digitize(n) {
        let charArr = n.toString().split('');
        let numArr = [];
        charArr.forEach((char) => {
            let num = parseInt(char, 10);
            numArr.push(num);
        });
        return numArr.reverse();
    }
